Machine Coding 1: Shopping Cart

Problem

Create a shopping cart where users can add/remove items and calculate total price.

Solution


const cart = [];

function addToCart(product) {
  const existing = cart.find(item => item.id === product.id);
  if (existing) {
    existing.qty++;
  } else {
    cart.push({ ...product, qty: 1 });
  }
}

function removeFromCart(id) {
  const index = cart.findIndex(item => item.id === id);
  if (index !== -1) {
    cart[index].qty--;
    if (cart[index].qty === 0) cart.splice(index, 1);
  }
}

function getTotal() {
  return cart.reduce((sum, item) => sum + item.price * item.qty, 0);
}

// Example
addToCart({ id: 1, name: "Phone", price: 20000 });
addToCart({ id: 1, name: "Phone", price: 20000 });
removeFromCart(1);

console.log(cart);
console.log("Total:", getTotal());
Interviewer checks: state handling, immutability discussion, edge cases.

Machine Coding 2: Infinite Scroll

Problem

Load data when user reaches bottom of page.

Solution


let page = 1;
let loading = false;

async function loadData() {
  if (loading) return;
  loading = true;

  const res = await fetch(
    'https://jsonplaceholder.typicode.com/posts?_limit=10&_page=' + page
  );
  const data = await res.json();

  const container = document.getElementById("list");
  data.forEach(item => {
    const div = document.createElement("div");
    div.textContent = item.title;
    container.appendChild(div);
  });

  page++;
  loading = false;
}

window.addEventListener("scroll", () => {
  if (
    window.innerHeight + window.scrollY >=
    document.body.offsetHeight - 100
  ) {
    loadData();
  }
});

loadData();
Interviewer checks: performance, debounce/throttle, API pagination.

Machine Coding 3: Authentication Flow

Problem

Implement login, token storage, protected route logic.

Solution


// Fake API
function login(username, password) {
  return new Promise((resolve, reject) => {
    if (username === "admin" && password === "1234") {
      resolve({ token: "fake-jwt-token" });
    } else {
      reject("Invalid credentials");
    }
  });
}

// Login handler
async function handleLogin() {
  try {
    const res = await login("admin", "1234");
    localStorage.setItem("token", res.token);
    console.log("Login success");
  } catch (err) {
    console.error(err);
  }
}

// Protected route
function isAuthenticated() {
  return !!localStorage.getItem("token");
}

function accessDashboard() {
  if (isAuthenticated()) {
    console.log("Welcome to Dashboard");
  } else {
    console.log("Redirect to Login");
  }
}

handleLogin();
accessDashboard();
Interviewer checks: auth concept, token storage, security discussion.
