Day 1: Introduction to JavaScript

JavaScript is a high-level, interpreted programming language used for web development. It can change HTML content, handle events, create animations, and interact with servers.

// Example
console.log("Hello, JavaScript!");

Practice: Open browser console and print your name using console.log

Day 2: Variables & Data Types

Learn var, let, const and types: number, string, boolean, undefined, null.

let name = "Nisha";
const pi = 3.14;
var age = 22;
console.log(name, pi, age);

Day 3: Operators & Expressions

Arithmetic, assignment, comparison, logical, and ternary operators.

let a = 5 + 3; // 8
let isAdult = age >= 18 ? "Yes" : "No";

Day 4: Conditionals

if(age >= 18) {
  console.log("Adult");
} else {
  console.log("Minor");
}

Day 5: Loops

for(let i=0;i<5;i++){
  console.log(i);
}
let j=0;
while(j<5){
  console.log(j);
  j++;
}

Day 8: Objects

Day 9: Advanced Functions

// Function expressions & arrow functions

Day 10: 'this' Keyword

let user = {
  name: "Nisha",
  greet() { console.log(this.name); }
};
user.greet();

Day 11: Scope & Hoisting

console.log(a); // undefined
var a = 10;

Day 12: Closures

function outer() {
  let count = 0;
  return function inner() {
    count++;
    console.log(count);
  }
}
let counter = outer();
counter(); // 1

Day 13: Async JavaScript

setTimeout(() => console.log("Hello after 1s"), 1000);

async function fetchData() {
  let response = await fetch("https://jsonplaceholder.typicode.com/users/1");
  let data = await response.json();
  console.log(data.name);
}
fetchData();

Day 20: Advanced JS Recap

async function fetchUser(id){
  try{
    let res = await fetch(`https://jsonplaceholder.typicode.com/users/${id}`);
    let data = await res.json();
    console.log(data.name);
  }catch(err){ console.log(err.message);}
}
fetchUser(1);
